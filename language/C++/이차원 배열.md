배열의 열 수(3) 는 타입의 일부이므로 컴파일 타임에 고정되어야 합니다.
```cpp
#include <iostream>

int main() {
    // 4행 3열짜리 2차원 배열을 동적 할당
    int (*a)[3] = new int[4][3];  // 타입: "3열짜리 배열을 가리키는 포인터"

    // 값 초기화
    for (int i = 0; i < 4; ++i) {
        for (int j = 0; j < 3; ++j) {
            a[i][j] = i * 10 + j;
        }
    }

    // 값 출력
    for (int i = 0; i < 4; ++i) {
        for (int j = 0; j < 3; ++j) {
            std::cout << a[i][j] << ' ';
        }
        std::cout << '\n';
    }

    // 메모리 해제
    delete[] a;

    return 0;
}
```
그냥 일차원 배열로 쓰는게 가독성이 좋다...
```cpp
#include <iostream>
int main() {
    int rows = 4;
    int cols = 3;

    // 1차원 배열로 전체 메모리 확보
    int* arr = new int[rows * cols];

    // 값 넣기 (2차원처럼)
    for (int i = 0; i < rows; ++i) {
        for (int j = 0; j < cols; ++j) {
            arr[i * cols + j] = i * 10 + j;
        }
    }

    // 출력 (2차원처럼)
    for (int i = 0; i < rows; ++i) {
        for (int j = 0; j < cols; ++j) {
            std::cout << arr[i * cols + j] << ' ';
        }
        std::cout << '\n';
    }

    delete[] arr;
    return 0;
}
```
