```bash
#!/bin/bash

echo "configure build output path"

KERNEL_TOP_PATH="$( cd "$(dirname "$0")" ; pwd -P)" # 파일포함한 전체 경로 표시  거기에 서 dirname은 디렉토리만 뽑고 cd한다 그리고 현재 경로 출력한 데이터를 변수에 저장
OUTPUT="$KERNEL_TOP_PATH/out" # output 경로를 변수에 저장
echo "$OUTPUT"

KERNEL=kernel8 # 커널 버전 지정
BUILD_LOG="$KERNEL_TOP_PATH/rpi_build_log.txt" #빌드 로그 txt파일에 저장

echo "move kernel source"
cd linux

echo "make defconfig" # 
make O=$OUTPUT bcm2711_defconfig

echo "kernel build"
make O=$OUTPUT Image modules dtbs -j4 2>&1 | tee $BUILD_LOGvi
```
