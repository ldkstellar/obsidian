[[Shell script 문법]]
```bash
#!/bin/bash

echo "configure build output path"

KERNEL_TOP_PATH="$( cd "$(dirname "$0")" ; pwd -P)" # 파일포함한 전체 경로 표시  거기에 서 dirname은 디렉토리만 뽑고 cd한다 그리고 현재 경로 출력한 데이터를 변수에 저장
OUTPUT="$KERNEL_TOP_PATH/out" # output 경로를 변수에 저장
echo "$OUTPUT"

KERNEL=kernel8 # 커널 버전 지정
BUILD_LOG="$KERNEL_TOP_PATH/rpi_build_log.txt" #빌드 로그 txt파일에 저장

echo "move kernel source"
cd linux

echo "make defconfig" # 설정 불러오기 
make O=$OUTPUT bcm2711_defconfig # O는 output diractory 지정

echo "kernel build"
make O=$OUTPUT Image modules dtbs -j4 2>&1 | tee $BUILD_LOGvi # 커널 바이너리 커널 모듈(드라이버) ,디바이스 treeblob 파일 4코어 사용해서 동시에 빌드하기 표준출력(1) 에러출력(2)리다이렉션 표준출력으로 함께 처리
# tee 명령은 출력내용을 화면에도 출력하면서 동시에 로그파일1에도 저장한다.
```
